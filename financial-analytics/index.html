Query 1 · Anomaly Detection
SELECT txn_id, amount, z_score,
  CASE WHEN z_score > 3.0
  THEN 'ANOMALY'
  ELSE 'NORMAL' END AS flag
FROM transactions_v2
WHERE txn_date = CURDATE()
ORDER BY z_score DESC;
⚡ Executed in 0.82s · 2.4M rows scanned
Query 2 · Hourly Volume Agg
SELECT
  DATE_FORMAT(txn_time,'%H:00'),
  COUNT(*) AS txn_count,
  SUM(amount) AS total_vol
FROM transactions_v2
GROUP BY HOUR(txn_time)
ORDER BY 1;
⚡ Executed in 1.24s · Index optimized
Query 3 · Duplicate Detection
SELECT account_id, amount,
  COUNT(*) AS dup_count
FROM transactions_v2
WHERE txn_time BETWEEN
  NOW() - INTERVAL 5 MINUTE AND NOW()
GROUP BY account_id, amount
HAVING dup_count > 1;
⚡ Executed in 0.44s · 12 duplicates found
